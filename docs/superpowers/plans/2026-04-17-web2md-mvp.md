# web2md MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `web2md` Go CLI that converts a web article into an offline Markdown note with locally downloaded article media.

**Architecture:** The CLI entrypoint in `cmd/root.go` validates flags and calls a small orchestration package. Fetching, parsing, downloading, converting, and progress reporting live in focused packages under `pkg/` so each behavior can be tested independently with local fixtures.

**Tech Stack:** Go, Cobra, go-readability, goquery, html-to-markdown, standard `net/http`, local fixture tests with `go test ./...`.

---

## File Structure

Create these files:

- `go.mod`: Go module metadata and library dependencies.
- `main.go`: Calls `cmd.Execute()`.
- `cmd/root.go`: Cobra command, flags, help text, exit behavior.
- `pkg/app/app.go`: Orchestrates fetch, parse, download, convert, and write.
- `pkg/fetcher/fetcher.go`: HTTP client with 30 second timeout, browser-like User-Agent, response checks.
- `pkg/parser/parser.go`: Readability extraction, media discovery, media URL resolution.
- `pkg/downloader/downloader.go`: Concurrent downloads, filename collision avoidance, tolerant and strict failure behavior.
- `pkg/converter/converter.go`: HTML-to-Markdown conversion and frontmatter injection.
- `pkg/progress/progress.go`: Docker-pull-style text progress events.
- `testdata/article.html`: Fixture article with clutter, images, and videos.
- `testdata/empty.html`: Fixture that fetches but yields no useful article body.
- Package test files beside their packages.
- `scripts/smoke-public-url.ps1`: Optional manual smoke test against a public URL.

package cmd

import (
	"fmt"
	"io"
	"os"

	"github.com/spf13/cobra"
)

type Options struct {
	Name   string
	Strict bool
}

func Execute() error {
	return NewRootCommand(os.Stdout, os.Stderr).Execute()
}

func NewRootCommand(stdout, stderr io.Writer) *cobra.Command {
	opts := &Options{}
	cmd := &cobra.Command{
		Use:          "web2md <URL>",
		Short:        "Convert a web article into an offline Markdown note",
		Args:         cobra.ExactArgs(1),
		SilenceUsage: true,
		RunE: func(cmd *cobra.Command, args []string) error {
			if opts.Name == "" {
				fmt.Fprintln(stderr, "错误：需要提供 -n 或 --name 指定输出文档名称。")
				_ = cmd.Help()
				return fmt.Errorf("missing required name")
			}
			fmt.Fprintf(stdout, "web2md planned run: url=%s name=%s strict=%v\n", args[0], opts.Name, opts.Strict)
			return nil
		},
	}
	cmd.SetOut(stdout)
	cmd.SetErr(stderr)
	cmd.Flags().StringVarP(&opts.Name, "name", "n", "", "Markdown 文件名，不含 .md 后缀")
	cmd.Flags().BoolVar(&opts.Strict, "strict", false, "资源下载失败时立即返回错误")
	return cmd
}

package fetcher

import (
	"fmt"
	"io"
	"net/http"
	"time"
)

type ClientConfig struct {
	Timeout time.Duration
}

type Client struct {
	httpClient *http.Client
}

type Page struct {
	URL  string
	Body []byte
}

func New(config ClientConfig) *Client {
	timeout := config.Timeout
	if timeout == 0 {
		timeout = 30 * time.Second
	}
	return &Client{httpClient: &http.Client{Timeout: timeout}}
}

func (c *Client) Fetch(rawURL string) (Page, error) {
	req, err := http.NewRequest(http.MethodGet, rawURL, nil)
	if err != nil {
		return Page{}, fmt.Errorf("错误：无法访问该网页，请检查网络连接或链接是否正确。")
	}
	req.Header.Set("User-Agent", "web2md/1.0 (+https://example.invalid/web2md)")

	resp, err := c.httpClient.Do(req)
	if err != nil {
		return Page{}, fmt.Errorf("错误：无法访问该网页，请检查网络连接或链接是否正确。")
	}
	defer resp.Body.Close()

	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		return Page{}, fmt.Errorf("错误：无法访问该网页，请检查网络连接或链接是否正确。")
	}
	body, err := io.ReadAll(resp.Body)
	if err != nil {
		return Page{}, fmt.Errorf("读取网页内容失败: %w", err)
	}
	return Page{URL: resp.Request.URL.String(), Body: body}, nil
}
```

package parser

import (
	"bytes"
	"net/url"
	"path"
	"strings"

	"github.com/PuerkitoBio/goquery"
	readability "github.com/go-shiori/go-readability"
)

type ResourceType string

const (
	ResourceImage ResourceType = "image"
	ResourceVideo ResourceType = "video"
)

type Resource struct {
	Type        ResourceType
	OriginalURL string
	ResolvedURL string
	Attr        string
}

type Result struct {
	HTML       string
	Title      string
	Resources  []Resource
	HasContent bool
}

func Parse(pageURL string, body []byte) (Result, error) {
	base, err := url.Parse(pageURL)
	if err != nil {
		return Result{}, err
	}
	article, err := readability.FromReader(bytes.NewReader(body), base)
	if err != nil {
		return Result{HasContent: false}, nil
	}
	html := strings.TrimSpace(article.Content)
	if html == "" {
		return Result{Title: article.Title, HasContent: false}, nil
	}
	doc, err := goquery.NewDocumentFromReader(strings.NewReader(html))
	if err != nil {
		return Result{}, err
	}

	resources := collectResources(doc, base)
	cleaned, err := doc.Find("body").Html()
	if err != nil || strings.TrimSpace(cleaned) == "" {
		cleaned, err = doc.Html()
	}
	if err != nil {
		return Result{}, err
	}
	return Result{HTML: cleaned, Title: article.Title, Resources: resources, HasContent: strings.TrimSpace(cleaned) != ""}, nil
}

func collectResources(doc *goquery.Document, base *url.URL) []Resource {
	var resources []Resource
	doc.Find("img[src]").Each(func(_ int, sel *goquery.Selection) {
		src, _ := sel.Attr("src")
		if resolved := resolve(base, src); resolved != "" {
			resources = append(resources, Resource{Type: ResourceImage, OriginalURL: src, ResolvedURL: resolved, Attr: "src"})
		}
	})
	doc.Find("video[src], video source[src]").Each(func(_ int, sel *goquery.Selection) {
		src, _ := sel.Attr("src")
		if resolved := resolve(base, src); resolved != "" && isDirectVideo(resolved) {
			resources = append(resources, Resource{Type: ResourceVideo, OriginalURL: src, ResolvedURL: resolved, Attr: "src"})
		}
	})
	return resources
}

func resolve(base *url.URL, raw string) string {
	u, err := url.Parse(strings.TrimSpace(raw))
	if err != nil || u.String() == "" {
		return ""
	}
	return base.ResolveReference(u).String()
}

func isDirectVideo(raw string) bool {
	u, err := url.Parse(raw)
	if err != nil {
		return false
	}
	ext := strings.ToLower(path.Ext(u.Path))
	switch ext {
	case ".mp4", ".webm", ".mov", ".m4v":
		return true
	default:
		return false
	}
}
```

package downloader

import (
	"fmt"
	"io"
	"net/http"
	"net/url"
	"os"
	"path"
	"path/filepath"
	"strings"
	"sync"
	"time"
)

type Resource struct {
	URL string
}

type Result struct {
	URL         string
	LocalPath   string
	Replacement string
	Failed      bool
	Err         error
}

type Config struct {
	Concurrency int
	Strict      bool
	Events      EventSink
}

type EventSink interface {
	EventName(name string, status string)
}

type Downloader struct {
	config Config
	client *http.Client
}

func New(config Config) *Downloader {
	if config.Concurrency <= 0 {
		config.Concurrency = 5
	}
	return &Downloader{config: config, client: &http.Client{Timeout: 30 * time.Second}}
}

func (d *Downloader) DownloadAll(assetDir string, resources []Resource) ([]Result, error) {
	if err := os.MkdirAll(assetDir, 0o755); err != nil {
		return nil, err
	}
	results := make([]Result, len(resources))
	jobs := make(chan int)
	var mu sync.Mutex
	var wg sync.WaitGroup
	var firstErr error
	used := map[string]bool{}

	for i := 0; i < d.config.Concurrency; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			for idx := range jobs {
				res := resources[idx]
				localName := d.reserveName(assetDir, res.URL, used, &mu)
				localAbs := filepath.Join(assetDir, localName)
				localRel := "./assets/" + filepath.ToSlash(localName)
				d.emit(localName, "Downloading")
				err := d.download(res.URL, localAbs)
				result := Result{URL: res.URL, LocalPath: localRel, Replacement: localRel}
				if err != nil {
					result.Failed = true
					result.Err = err
					result.Replacement = res.URL
					d.emit(localName, "Failed")
					if d.config.Strict {
						mu.Lock()
						if firstErr == nil {
							firstErr = err
						}
						mu.Unlock()
					}
				} else {
					d.emit(localName, "Complete")
				}
				results[idx] = result
			}
		}()
	}

	for i := range resources {
		mu.Lock()
		stop := firstErr != nil
		mu.Unlock()
		if stop {
			break
		}
		jobs <- i
	}
	close(jobs)
	wg.Wait()

	mu.Lock()
	err := firstErr
	mu.Unlock()
	if err != nil {
		return results, err
	}
	return results, nil
}

func (d *Downloader) emit(name string, status string) {
	if d.config.Events != nil {
		d.config.Events.EventName(name, status)
	}
}

func (d *Downloader) reserveName(assetDir, rawURL string, used map[string]bool, mu *sync.Mutex) string {
	name := filenameFromURL(rawURL)
	ext := filepath.Ext(name)
	base := strings.TrimSuffix(name, ext)
	mu.Lock()
	defer mu.Unlock()
	candidate := name
	for i := 1; used[candidate] || fileExists(filepath.Join(assetDir, candidate)); i++ {
		candidate = fmt.Sprintf("%s_%d%s", base, i, ext)
	}
	used[candidate] = true
	return candidate
}

func (d *Downloader) download(rawURL, target string) error {
	resp, err := d.client.Get(rawURL)
	if err != nil {
		return err
	}
	defer resp.Body.Close()
	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		return fmt.Errorf("resource returned status %d", resp.StatusCode)
	}
	file, err := os.Create(target)
	if err != nil {
		return err
	}
	defer file.Close()
	_, err = io.Copy(file, resp.Body)
	return err
}

func filenameFromURL(raw string) string {
	u, err := url.Parse(raw)
	if err != nil {
		return "resource.bin"
	}
	name := path.Base(u.Path)
	if name == "." || name == "/" || name == "" {
		return "resource.bin"
	}
	return name
}

func fileExists(path string) bool {
	_, err := os.Stat(path)
	return err == nil
}

package converter

import (
	"fmt"
	"strings"
	"time"

	md "github.com/JohannesKaufmann/html-to-markdown"
)

type Document struct {
	OriginalURL string
	FetchDate   time.Time
	HTML        string
	HasContent  bool
}

func Convert(doc Document) (string, error) {
	body := ""
	if doc.HasContent || strings.TrimSpace(doc.HTML) != "" {
		converter := md.NewConverter("", true, nil)
		converted, err := converter.ConvertString(doc.HTML)
		if err != nil {
			return "", err
		}
		body = strings.TrimSpace(converted)
	} else {
		body = "原文链接：" + doc.OriginalURL
	}

	frontmatter := fmt.Sprintf("---\noriginal_url: %s\nfetch_date: %s\n---\n\n",
		doc.OriginalURL,
		doc.FetchDate.Format("2006-01-02 15:04:05"),
	)
	return frontmatter + body + "\n", nil
}

import (

)

type Event struct {
	Name   string
	Status string
}

type Renderer struct {
	out io.Writer
	mu  sync.Mutex
}

func NewRenderer(out io.Writer) *Renderer {
	return &Renderer{out: out}
}

func (r *Renderer) Event(event Event) {
	r.EventName(event.Name, event.Status)
}

func (r *Renderer) EventName(name string, status string) {
	r.mu.Lock()
	defer r.mu.Unlock()
	fmt.Fprintf(r.out, "%s: %s\n", name, status)

func RewriteResources(html string, replacements map[string]string) (string, error) {
	doc, err := goquery.NewDocumentFromReader(strings.NewReader(html))
	if err != nil {
		return "", err
	}
	doc.Find("img[src], video[src], video source[src]").Each(func(_ int, sel *goquery.Selection) {
		src, _ := sel.Attr("src")
		if replacement, ok := replacements[src]; ok {
			sel.SetAttr("src", replacement)
		}
	})
	body, err := doc.Find("body").Html()
	if err != nil || strings.TrimSpace(body) == "" {
		body, err = doc.Html()
	}
	return body, err
}

package app

import (
	"os"
	"path/filepath"
	"time"

	"webtomd/pkg/converter"
	"webtomd/pkg/downloader"
	"webtomd/pkg/fetcher"
	"webtomd/pkg/parser"
)

type Config struct {
	URL    string
	Name   string
	WorkDir string
	Strict bool
}

func Run(config Config) error {
	if config.WorkDir == "" {
		wd, err := os.Getwd()
		if err != nil {
			return err
		}
		config.WorkDir = wd
	}

	page, err := fetcher.New(fetcher.ClientConfig{Timeout: 30 * time.Second}).Fetch(config.URL)
	if err != nil {
		return err
	}
	parsed, err := parser.Parse(page.URL, page.Body)
	if err != nil {
		return err
	}

	html := parsed.HTML
	if parsed.HasContent && len(parsed.Resources) > 0 {
		resources := make([]downloader.Resource, 0, len(parsed.Resources))
		for _, res := range parsed.Resources {
			resources = append(resources, downloader.Resource{URL: res.ResolvedURL})
		}
		results, err := downloader.New(downloader.Config{Concurrency: 5, Strict: config.Strict}).DownloadAll(filepath.Join(config.WorkDir, "assets"), resources)
		if err != nil {
			return err
		}
		replacements := map[string]string{}
		for i, res := range parsed.Resources {
			replacements[res.OriginalURL] = results[i].Replacement
			replacements[res.ResolvedURL] = results[i].Replacement
		}
		html, err = parser.RewriteResources(html, replacements)
		if err != nil {
			return err
		}
	}

	md, err := converter.Convert(converter.Document{
		OriginalURL: page.URL,
		FetchDate:   time.Now(),
		HTML:        html,
		HasContent:  parsed.HasContent,
	})
	if err != nil {
		return err
	}
	return os.WriteFile(filepath.Join(config.WorkDir, config.Name+".md"), []byte(md), 0o644)
}
